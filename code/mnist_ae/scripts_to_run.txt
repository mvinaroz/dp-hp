#!/bin/bash

# list of commands to run for reproduction purposes

### digit MNIST

# DP-MERF
python3 train_dp_autoencoder_directly.py --log-name dp_merf_digits --data digits
python3 synth_data_benchmark.py --data-log-name dp_merf_digits --data digits

# DP-MERF+AE (first autoencoder training, then generator training, then evaluation
python3 train_dp_autoencoder.py --log-name dp_merf_ae_digits --data digits
python3 train_dp_generator.py --ae-load-dir logs/ae/dp_merf_ae_digits/ --log-name d0_1_0 --data digits
python3 synth_data_benchmark.py --data-log-name dp_merf_ae_digits --data digits

# DP-CGAN
python3 dp_cgan_reference.py --data-save-str dpcgan_digits
python3 synth_data_benchmark.py --data-log-name dpcgan_digits --data digits

### fashion MNIST
# DP-MERF
python3 train_dp_autoencoder_directly.py --log-name dp_merf_fashion --data fashion
python3 synth_data_benchmark.py --data-log-name dp_merf_fashion --data fashion

# DP-MERF+AE
python3 train_dp_autoencoder.py --log-name dp_merf_ae_fashion --data fashion
python3 train_dp_generator.py --ae-load-dir logs/ae/dp_merf_ae_fashion/ --log-name f0_1_0 --data fashion
python3 synth_data_benchmark.py --data-log-name dp_merf_ae_fashion --data fashion

# DP-CGAN
python3 dp_cgan_reference.py --data-save-str dpcgan_fashion --data fashion
python3 synth_data_benchmark.py --data-log-name dpcgan_fashion --data fashion
